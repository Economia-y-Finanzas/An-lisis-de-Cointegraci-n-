# Errores estándar y estadísticos t de los coeficientes de ajuste (α) del VECM.
resids <- resid(vecm.r1$rlm)
sigma <- crossprod(resids) / nrow(resids)
alpha.se <- sqrt(solve(crossprod(cbind(vecm@ZK %*% beta, vecm@Z1)))[1, 1] * diag(sigma))
names(alpha.se) <-  c('aaa', 'bbb')
alpha.t <- alpha / alpha.se
alpha.t

# Errores estándar y los estadísticos t del vector de cointegración (β) del VECM.
